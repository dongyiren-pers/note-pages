# 解决哈希冲突的方法和 Java 中的实现

快手腾讯PCG
  • 开放定址法：发生哈希冲突时，使用某种探测技术寻找下一个空的散列地址。Java 中的 ThreadLocalMap 基于此解决冲突。
  • 链地址法：发生哈希冲突时，被分配到同一个索引上的多个节点通过单向链表连接起来。Java 中的 HashMap 和 ConcurrentHashMap 基于此解决冲突。
  • 再哈希法：发生哈希冲突时，使用不同的哈希函数重新散列。
  • 建立公共溢出区：发生哈希冲突时，将与基本表发生冲突的元素填入溢出表。

# 红黑树的特征

京东字节
  • 每个节点要么为黑色要么为红色。
  • 根节点为 黑色。
  • 叶子节点（Nil 空节点）为 黑色。
  • 红色节点的孩子都为 黑色 节点。
  • 任一节点到叶子节点的 黑色 节点的数量一致。

// 统一奇偶回文串
public String handle(String s) {
    StringBuilder sb = new StringBuilder();
    sb.append("$");
    for (int i = 0; i < s.length(); i++)
        sb.append("#").append(s.charAt(i));
    sb.append("#");
    sb.append("^");
    return sb.toString();
}

// Manacher 算法
public int[] manacher(String s) {
    int n = s.length();
    int[] p = new int[n];
    int l = 1, r = 1;
    for (int i = 1; i < n - 1; i++) {
        p[i] = Math.max(0, Math.min(r - i, p[l + r - i]));
        while (s.charAt(i - p[i] - 1) == s.charAt(i + p[i] + 1))
            p[i]++;
        if (i + p[i] > r) {
            l = i - p[i];
            r = i + p[i];
        }
    }
    return p;
}

public String longestPalindrome(String s) {
    int[] p = manacher(handle(s));
    int maxIdx = 1;
    for (int i = 1; i < p.length - 1; i++) {
        if (p[i] > p[maxIdx])
            maxIdx = i;
    }
    int start = (maxIdx - p[maxIdx] - 1) / 2;
    return s.substring(start, start + p[maxIdx]);
}

# KMP

详见 Knuth-Morris-Pratt 算法。

public int kmp(String s, String p) {
    int n = s.length(), m = p.length();
    int[] pi = new int[m];
    for (int i = 1, j = 0; i < m; i++) {
        while (j > 0 && p.charAt(i) != p.charAt(j)) j = pi[j - 1];
        if (p.charAt(i) == p.charAt(j)) j++;
        pi[i] = j;
    }
    for (int i = 0, j = 0; i < n; i++) {
        while (j > 0 && s.charAt(i) != p.charAt(j)) j = pi[j - 1];
        if (s.charAt(i) == p.charAt(j)) j++;
        if (j == m) return i - m + 1;
    }
    return -1;
}

# 堆排序

详见 堆排序。

// 大顶堆向下调整
void shiftDown(int[] a, int start, int end) {
    int current = start;            		// 当前节点位置
    int left = 2 * current + 1;       		// 左孩子的位置
    while (left <= end) {
        // left 是左孩子，left + 1 是右孩子
        if (left < end && a[left] < a[left + 1])
            left++;							// 左右两孩子中选择较大者
        if (a[current] >= a[left]) break;	// 调整结束
        swap(a, current, left);				// 交换值
        current = left;
        left = 2 * current + 1;
    }
}
// 堆排序（升序）：交换数据，将 a[1] 和 a[n] 交换，使 a[n] 是 a[1...n] 中的最大值；
// 然后将 a[1...n - 1] 重新调整为大顶堆。
void heapSort(int[] a, int n) {
    for (int i = n / 2 - 1; i >= 0; i--)		// 从 (n / 2 - 1) --> 0 逐次遍历。
        shiftDown(a, i, n - 1);
    for (int i = n - 1; i > 0; i--){ 
        // 交换 a[0] 和 a[i]。交换后，a[i] 是 a[0...i] 中最大的。
        swap(a, 0, i);
        // 调整 a[0...i -1 ] ，使得 a[0...i - 1] 仍然是一个最大堆。
        shiftDown(a, 0, i - 1);
    }
}

# 快速排序

详见 快速排序。

// 朴素快排，首位为主轴
public int[] quickSortSimple(int[] arr) {
    if (arr.length < 2) return arr;
    quickSortSimple(arr, 0, arr.length - 1);
    return arr;
}

private void quickSortSimple(int[] arr, int left, int right) {
    if (left < right) {
        int pivot = partition(arr, left, right);
        quickSortSimple(arr, left, pivot - 1);
        quickSortSimple(arr, pivot + 1, right);
    }
}

//　partition 方法
private int partition(int[] arr, int left, int right) {
    int pivot = left, index = pivot + 1;
    // 注意此时 right 是坐标，要执行到最后一个元素
    for (int i = index; i <= right; i++) {
        if (arr[i] < arr[pivot]) {
            swap(arr, index, i);
            index++;
        }
    }
    // 最后一个小于主轴元素的元素下标是 index - 1
    swap(arr, pivot, index - 1);
    return index - 1;
}
